# 图表展示模块设计

## 1. 模块概述

图表展示模块是肾功能监测应用的重要组成部分，用于可视化用户的肾功能指标数据，帮助用户直观地了解自己的健康状况变化趋势。该模块实现了折线图、柱状图和环形图等多种图表类型，并支持缩放、滑动、点击查看详情等交互功能。

## 2. 技术选型

- **图表库**：MPAndroidChart 3.1.0+
- **UI框架**：Jetpack Compose
- **数据处理**：Kotlin Coroutines
- **状态管理**：ViewModel + LiveData

## 4. 核心图表组件

### 4.1 折线图（LineChart）

用于展示eGFR、肌酐等连续指标的趋势变化。

```kotlin
@Composable
fun LineChart(measurements: List<Measurement>, indicatorType: IndicatorType) {
    val context = LocalContext.current
    
    AndroidView(
        factory = {
            LineChart(it).apply {
                setupLineChart(this, measurements, indicatorType)
            }
        },
        modifier = Modifier
            .fillMaxWidth()
            .height(300.dp)
    )
}

private fun setupLineChart(chart: LineChart, measurements: List<Measurement>, indicatorType: IndicatorType) {
    // 配置图表
    chart.apply {
        // 基本设置
        description.isEnabled = false
        setTouchEnabled(true)
        setDrawGridBackground(false)
        setDragEnabled(true)
        setScaleEnabled(true)
        setPinchZoom(true)
        setExtraOffsets(10f, 10f, 10f, 10f)
        
        // 图例
        legend.apply {
            verticalAlignment = Legend.LegendVerticalAlignment.TOP
            horizontalAlignment = Legend.LegendHorizontalAlignment.RIGHT
            orientation = Legend.LegendOrientation.VERTICAL
            setDrawInside(false)
            textSize = 12f
        }
        
        // X轴
        xAxis.apply {
            position = XAxis.XAxisPosition.BOTTOM
            setDrawGridLines(false)
            granularity = 1f
            valueFormatter = DateValueFormatter(measurements.map { it.timestamp })
        }
        
        // Y轴
        axisLeft.apply {
            setDrawGridLines(true)
            granularity = 1f
            labelCount = 5
        }
        axisRight.isEnabled = false
        
        // 数据集
        val entries = measurements.mapIndexed { index, measurement ->
            val value = getIndicatorValue(measurement, indicatorType)
            Entry(index.toFloat(), value)
        }
        
        val dataSet = LineDataSet(entries, getIndicatorTitle(indicatorType)).apply {
            lineWidth = 2f
            circleRadius = 4f
            setCircleColor(getIndicatorColor(indicatorType))
            color = getIndicatorColor(indicatorType)
            valueTextSize = 10f
            setDrawValues(false)
            mode = LineDataSet.Mode.CUBIC_BEZIER
            cubicIntensity = 0.2f
        }
        
        // 添加参考线
        addReferenceLines(chart, indicatorType)
        
        // 设置数据
        data = LineData(dataSet)
        invalidate()
    }
}
```

### 4.3 环形图（PieChart）

用于展示eGFR相对正常下限的比例。

```kotlin
@Composable
fun GFRRingChart(egfr: Float) {
    val context = LocalContext.current
    val normalGFR = 90f // 正常eGFR下限
    val percentage = minOf(egfr / normalGFR, 1f)
    
    AndroidView(
        factory = {
            PieChart(it).apply {
                setupRingChart(this, percentage)
            }
        },
        modifier = Modifier
            .size(200.dp)
            .align(Alignment.Center)
    )
}

private fun setupRingChart(chart: PieChart, percentage: Float) {
    // 配置图表
    chart.apply {
        // 基本设置
        description.isEnabled = false
        setTouchEnabled(false)
        setDrawHoleEnabled(true)
        holeRadius = 60f
        transparentCircleRadius = 65f
        setDrawCenterText(true)
        centerText = "${(percentage * 100).toInt()}%"
        setCenterTextSize(24f)
        
        // 图例
        legend.isEnabled = false
        
        // 数据集
        val entries = listOf(
            PieEntry(percentage, "当前eGFR"),
            PieEntry(1f - percentage, "剩余容量")
        )
        
        val dataSet = PieDataSet(entries, "").apply {
            colors = listOf(
                Color.Green,
                Color.LightGray
            )
            valueTextSize = 12f
            setDrawValues(false)
            sliceSpace = 2f
        }
        
        // 设置数据
        data = PieData(dataSet)
        invalidate()
    }
}
```

## 5. 数据适配器

### 5.1 日期值格式化器

```kotlin
class DateValueFormatter(private val timestamps: List<Long>) : ValueFormatter() {
    override fun getFormattedValue(value: Float): String {
        val index = value.toInt()
        if (index < 0 || index >= timestamps.size) return ""
        
        val date = Date(timestamps[index])
        val formatter = SimpleDateFormat("MM-dd", Locale.getDefault())
        return formatter.format(date)
    }
}
```

### 6.3 区间参考线

```kotlin
private fun addReferenceLines(chart: BarLineChartBase<*>, indicatorType: IndicatorType) {
    val (minNormal, maxNormal) = getNormalRange(indicatorType)
    
    // 添加正常范围下限参考线
    if (minNormal != null) {
        val limitLineMin = LimitLine(minNormal, "正常下限").apply {
            lineColor = Color.Green
            lineWidth = 1f
            textColor = Color.Green
            textSize = 10f
        }
        chart.axisLeft.addLimitLine(limitLineMin)
    }
    
    // 添加正常范围上限参考线
    if (maxNormal != null) {
        val limitLineMax = LimitLine(maxNormal, "正常上限").apply {
            lineColor = Color.Green
            lineWidth = 1f
            textColor = Color.Green
            textSize = 10f
        }
        chart.axisLeft.addLimitLine(limitLineMax)
    }
}

private fun getNormalRange(type: IndicatorType): Pair<Float?, Float?> {
    return when (type) {
        IndicatorType.EGFR -> Pair(60f, 120f)
```

## 7. 性能优化

### 7.1 数据处理优化

```kotlin
// 处理大量数据时的优化
fun processChartData(measurements: List<Measurement>, maxPoints: Int = 100): List<Measurement> {
    if (measurements.size <= maxPoints) return measurements
    
    val step = measurements.size / maxPoints
    val result = mutableListOf<Measurement>()
    
    for (i in 0 until measurements.size step step) {
        result.add(measurements[i])
    }
    
    // 确保包含最后一个数据点
    if (result.lastOrNull() != measurements.lastOrNull()) {
        result.add(measurements.last())
    }
    
    return result
}
```

```kotlin
// 在Compose中使用AndroidView时的内存管理
@Composable
fun OptimizedChart(measurements: List<Measurement>, indicatorType: IndicatorType) {
    val processedData = remember(measurements) {
        processChartData(measurements)
    }
    
    AndroidView(
        factory = {
            when (indicatorType) {
                IndicatorType.EGFR, IndicatorType.CREATININE, IndicatorType.BUN -> {
                    LineChart(it).apply {
                        setupLineChart(this, processedData, indicatorType)
                    }
                }
                IndicatorType.PROTEINURIA, IndicatorType.URINE_VOLUME -> {
                    BarChart(it).apply {
                        setupBarChart(this, processedData, indicatorType)
                    }
                }
                else -> {
                    LineChart(it).apply {
                        setupLineChart(this, processedData, indicatorType)
                    }
                }
            }
        },
        modifier = Modifier
            .fillMaxWidth()
            .height(300.dp),
        update = { chart ->
            when (chart) {
                is LineChart -> setupLineChart(chart, processedData, indicatorType)
                is BarChart -> setupBarChart(chart, processedData, indicatorType)
            }
        }
    )
}
```
