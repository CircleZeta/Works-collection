# 多设备与用户管理设计

## 1. 概述

多设备与用户管理模块是肾功能监测应用的重要组成部分，用于管理多个蓝牙设备和用户档案。该模块实现了设备扫描、绑定、切换，以及用户档案的创建、编辑和切换等功能，确保应用能够支持多用户使用同一设备，或同一用户使用多个设备的场景。

### 2.1 设备扫描与发现

```kotlin
class DeviceScanner(private val context: Context, private val bleManager: BleManager) {
    private var isScanning = false
    private val discoveredDevices = mutableListOf<BleDevice>()
    private val scanCallbacks = mutableListOf<(List<BleDevice>) -> Unit>()
    
    // 开始扫描
    fun startScan(callback: (List<BleDevice>) -> Unit) {
        if (isScanning) return
        
        isScanning = true
        discoveredDevices.clear()
        scanCallbacks.add(callback)
        
        bleManager.startScan { device ->
            if (!discoveredDevices.any { it.address == device.address }) {
                discoveredDevices.add(device)
                notifyCallbacks()
            }
        }
        
        // 扫描超时
        Handler(Looper.getMainLooper()).postDelayed({
            stopScan()
        }, 10000) // 10秒后停止扫描
    }
    
    // 停止扫描
    fun stopScan() {
        if (!isScanning) return
        
        isScanning = false
        bleManager.stopScan()
        notifyCallbacks()
        scanCallbacks.clear()
    }
    
    // 通知回调
    private fun notifyCallbacks() {
        scanCallbacks.forEach { it(discoveredDevices) }
    }
    
    // 获取已发现的设备
    fun getDiscoveredDevices(): List<BleDevice> {
        return discoveredDevices
    }
}
```

### 2.2 设备绑定与管理

```kotlin
class DeviceManager(private val context: Context, private val deviceRepository: DeviceRepository) {
    // 绑定设备
    suspend fun bindDevice(bleDevice: BleDevice) {
        val device = Device(
            name = bleDevice.name,
            address = bleDevice.address,
            bondState = bleDevice.device.bondState,
            lastConnected = System.currentTimeMillis(),
            isDefault = false
        )
        
        // 检查设备是否已存在
        val existingDevice = deviceRepository.getDeviceByAddress(bleDevice.address)
        if (existingDevice != null) {
            // 更新设备信息
            deviceRepository.updateDevice(existingDevice.copy(
                name = bleDevice.name,
                bondState = bleDevice.device.bondState,
                lastConnected = System.currentTimeMillis()
            ))
        } else {
            // 插入新设备
            deviceRepository.insertDevice(device)
        }
    }
    
    // 解绑设备
    suspend fun unbindDevice(device: Device) {
        deviceRepository.deleteDevice(device)
    }
    
    // 设置默认设备
    suspend fun setDefaultDevice(address: String) {
        deviceRepository.setDefaultDevice(address)
    }
    
    // 获取默认设备
    suspend fun getDefaultDevice(): Device? {
        return deviceRepository.getDefaultDevice()
    }
    
    // 获取所有设备
    fun getAllDevices(): Flow<List<Device>> {
        return deviceRepository.getAllDevices()
    }
    
    // 连接设备
    fun connectDevice(bleDevice: BleDevice, callback: BleGattCallback) {
        bleManager.connect(bleDevice, callback)
    }
    
    // 断开设备连接
    fun disconnectDevice(bleDevice: BleDevice) {
        bleManager.disconnect(bleDevice)
    }
}
```

```kotlin
class DeviceStateMonitor(private val context: Context) {
    private val bluetoothManager: BluetoothManager by lazy {
        context.getSystemService(Context.BLUETOOTH_SERVICE) as BluetoothManager
    }
    
    // 监控设备连接状态
    fun monitorDeviceState(address: String): Flow<Int> {
        return callbackFlow { 
            val broadcastReceiver = object : BroadcastReceiver() {
                override fun onReceive(context: Context?, intent: Intent?) {
                    if (intent?.action == BluetoothDevice.ACTION_BOND_STATE_CHANGED) {
                        val device = intent.getParcelableExtra<BluetoothDevice>(BluetoothDevice.EXTRA_DEVICE)
                        if (device?.address == address) {
                            val bondState = intent.getIntExtra(BluetoothDevice.EXTRA_BOND_STATE, BluetoothDevice.BOND_NONE)
                            trySend(bondState)
                        }
                    }
                }
            }
            
            context.registerReceiver(broadcastReceiver, IntentFilter(BluetoothDevice.ACTION_BOND_STATE_CHANGED))
            
            awaitClose {
                context.unregisterReceiver(broadcastReceiver)
            }
        }
    }
    
    // 检查设备是否已绑定
    fun isDeviceBonded(address: String): Boolean {
        val device = bluetoothManager.adapter.getRemoteDevice(address)
        return device.bondState == BluetoothDevice.BOND_BONDED
    }
    
    // 绑定设备
    fun bondDevice(address: String): Boolean {
        val device = bluetoothManager.adapter.getRemoteDevice(address)
        return device.createBond()
    }
}
```

```kotlin
class UserSwitcher(private val userManager: UserManager, private val appSettingsRepository: AppSettingsRepository) {
    // 切换用户
    suspend fun switchUser(userId: Long) {
        // 设置默认用户
        userManager.setDefaultUser(userId)
        
        // 保存当前用户ID到设置
        appSettingsRepository.saveUserId(userId.toInt())
    }
    
    // 获取当前用户
    suspend fun getCurrentUser(): User? {
        val userId = appSettingsRepository.getUserId().toLong()
        if (userId == -1L) {
            return userManager.getDefaultUser()
        }
        return userManager.getUserById(userId)
    }
    
    // 初始化用户
    suspend fun initializeUser() {
        val currentUser = getCurrentUser()
        if (currentUser == null) {
            // 创建默认用户
            val defaultUser = User(
                name = "默认用户",
                age = 30,
                gender = "男",
                height = 170f,
                weight = 60f,
                medicalHistory = "",
                isDefault = true
            )
            val userId = userManager.createUser(defaultUser)
            switchUser(userId)
        }
    }
}
```
